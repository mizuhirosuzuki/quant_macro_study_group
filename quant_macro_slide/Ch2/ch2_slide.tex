\begin{frame}\frametitle{2期間・3期間の効用最大化問題を通じて、数値計算の基本的な概念を導入}
  \begin{enumerate}
      \item 離散近似とグリッド
      \item \textcolor{red}{最適化問題を解く}
        \begin{itemize}
            \item 制御変数を離散として最適化（グリッドサーチ）
            \item 制御変数を連続として最適化
            \item 1階条件を使った最適化
        \end{itemize}
      \item \textcolor{red}{内生的グリッド法}
      \item \textcolor{red}{3期間モデルと近似}
      \item リスク（状態変数として確率的な生産性を導入）
  \end{enumerate}
\end{frame}

\begin{frame}\frametitle{2期間のベンチマークモデル}
  \begin{itemize}
      \item 2期間の消費・貯蓄行動のモデル
        \begin{align*}
          \max_{c_1, c_2, a} &U(c_1, c_2) = u(c_1) + \beta u(c_2) \\
          \text{subject to} \quad
          &c_1 + a = w \\
          &c_2 = (1 + r) a
        \end{align*}
      \item 最適化の一階条件 (オイラー方程式)：
        $$
        u'(c_1) = \beta (1 + r) u'(c_2)
        $$
      \item ゴール：ある所得$w$のもとでの個人の貯蓄関数 $a = g(w)$ を求めること
        \begin{itemize}
          \item 上の2つの制約式から、$a = g(w)$がわかれば、$c_1$、$c_2$も$w$の関数として表せる
          \item （3つの変数を効用最大化のために選択できて、2つの制約があるので、1つの制御変数の関数がわかればほかの制御変数の関数もわかる）
        \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{最適化の方法}
  \begin{itemize}
      \item 離散化された$w$の各グリッド$w_i$において最適な$a$を求めることを考える
        \bigskip
      \item グリッドサーチ：制御変数（$a$）を離散化して、各$w_i$上でそれぞれの$a$を選んだときの目的関数の値を求め、目的関数を最大化するものを選びそれを$g(w_i)$とする
      \item （制御変数を連続として扱う）最適化： 各$w_i$上で、目的関数を最大化するような$a$を求め、それを$g(w_i)$とする
      \item 一階条件を使った最適化（求根法）：最適化の一階条件を使って以下のように残差関数を定義して
        $$
          R(w_i; a) \equiv \beta (1 + r) \frac{u'((1 + r) a)}{u'(w_i - a)} - 1
        $$
        各$w_i$上で、この残差関数を0にするような$a$を求め、それを$g(w_i)$とする
        \bigskip
      \item 計算時間の比較：
        \begin{itemize}
          \item グリッドサーチ (4.613 μs)
          \item 最適化 (14.291 μs)
          \item 一階条件を使った最適化 (11.500 μs)
        \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{射影法は政策関数全体をパラメトリックに近似する手法}
  \begin{itemize}
      \item 政策変数を、例えば多項式で近似する：
        $$
        a = g(w) \approx \hat{g}(w; \bm{\theta}) = \sum_{m=0}^M \theta_m w^m
        $$
      \item オイラー方程式から以下の残差関数を考えて、これが各$w$で0に近くなる$\bm{\theta}$を求める：
        $$
          R(w; \bm{\theta}) \equiv \beta (1 + r) \frac{u'((1 + r) \hat{g}(w; \bm{\theta}))}{u'(w - \hat{g}(w; \bm{\theta}))} - 1
        $$
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{3期間モデル：どうして近似が必要なのか}
  \begin{itemize}
      \item 以下の3期間での消費・貯蓄モデルを考える：
        \begin{align*}
          %\max_{c_1, c_2, c_3, a_2, a_3} &U(c_1, c_2, c_3) = \frac{c_1^{1 - \gamma}}{1 - \gamma} + \beta \frac{c_2^{1 - \gamma}}{1 - \gamma}+ \beta^2 \frac{c_3^{1 - \gamma}}{1 - \gamma} \\
          \max_{c_1, c_2, c_3, a_2, a_3} &U(c_1, c_2, c_3) = u'(c_1) + \beta u'(c_2) + \beta^2 u'(c_3) \\
          \text{subject to} \quad
          &c_1 + a_2 = y_1 + (1 + r) a_1 \\
          &c_2 + a_3 = y_2 + (1 + r) a_2 \\
          &c_3 = y_3 + (1 + r) a_3 
        \end{align*}
        \begin{itemize}
            \item 各期の所得$y$は状態変数ではなく外生的なパラメタ
        \end{itemize}
        \bigskip
      \item オイラー方程式（最適化の一階条件）は以下の通り:
        \begin{align*}
          u'(c_1) &= \beta (1 + r) u'(c_2) \\
          u'(c_2) &= \beta (1 + r) u'(c_3)
        \end{align*}
      \item ゴール：
        2期目から3期目にかけての政策関数 $a_3 = g_2(a_2)$と
        1期目から2期目にかけての政策関数 $a_2 = g_1(a_1)$を求めること
  \end{itemize}
\end{frame}

\begin{frame}\frametitle{3期間モデル：どうして近似が必要なのか}
  \begin{itemize}
    \item 2期目から3期目にかけての政策関数$a_3 = g_2(a_2)$ は2つ目のオイラー方程式を求根法で解けば求められる：
      $$
      u'(y_2 + (1 + r) a_{2, i} - a_3) = \beta (1 + r) u'(y_3 + (1 + r) a_3)
      $$
    \item 1期目から2期目にかけての政策関数$a_2 = g_1(a_1)$ は1つ目のオイラー方程式を求根法で解けば求められる、けど\dots：
      $$
      u'(y_1 + (1 + r) a_{1, i} - a_2) = \beta (1 + r) u'(y_2 + (1 + r) a_2 \textcolor{red}{- g_2(a_2)})
      $$
      \begin{itemize}
        \item どの$a_2$がこれを満たすかを考えたいとき、$g_2$はグリッド$\{a_{2, j}\}_{i=1}^N$上だけでなく、すべての（連続的な）$a_2$上で評価可能でないといけない
        \item 例えば「うーん、$a_{2, k}$だと左辺のほうが大きくなって$a_{2, k+1}$だと右辺のほうが大きいから、$g_1(a_{1, i})$は$a_{2, k}$と$a_{2, k+1}$の間にあるのはわかるんだけど、正確な値はわからないなー」では困る
        \item グリッド$\{a_{2, j}\}_{i=1}^N$ 以外の点でも$g_2$を評価できるように、内挿法によって$\{g_2(a_{2, j})\}_{i=1}^N$から$g_2(a_2)$を近似する
      \end{itemize}
    \item 近似された$g_2$をもとに各$a_{1, i}$でオイラー方程式を満たす$a_2$を求根法で求め、$g_1(a_{1, j})$とする
  \end{itemize}
\end{frame}
